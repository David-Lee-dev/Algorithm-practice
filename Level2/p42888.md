users = dict()
seq = 0


def solution(record):
    global seq

    for r in record:
        tmp_string = r.split()
        behave = tmp_string[0]
        user = tmp_string[1]
        if len(tmp_string) > 2:
            nick = tmp_string[2]
        else:
            nick = ""

        if behave == "Enter":
            if users.get(user):
                users[user]["nick"] = nick
                users[user]["orders"].append([behave, seq])
                seq += 1
            else:
                users[user] = {
                    "nick": nick,
                    "orders": [[behave, seq]]
                }
                seq += 1
        elif behave == "Leave":
            users[user]["orders"].append([behave, seq])
            seq += 1
        elif behave == "Change":
            users[user]["nick"] = nick

    user_list = users.keys()
    logs = [""] * seq

    for user in user_list:
        nick = users[user]["nick"]
        orders = users[user]["orders"]
        for order in orders:
            behave = order[0]
            if behave == "Enter":
                logs[order[1]] = f'{nick}님이 들어왔습니다.'
            elif behave == "Leave":
                logs[order[1]] = f'{nick}님이 나갔습니다.'

    return logs

print(solution(["Enter uid1234 Muzi", "Enter uid4567 Prodo", "Leave uid1234", "Enter uid1234 Prodo", "Change uid4567 Ryan"]))
