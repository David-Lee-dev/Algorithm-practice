T = int(input())

def BF(n):
    global cnt
    global num

    if(n > num):
        return

    if(n == num):
        cnt += 1
        return

    BF(n + 1)
    BF(n + 2)
    BF(n + 3)

for _ in range(T):
    cnt = 0
    num = int(input())

    BF(1)
    BF(2)
    BF(3)

    print(cnt)
